-- Hàm để tạo ESP
local function createESP(player)
    -- Tạo tên người chơi
    local nameLabel = Instance.new("TextLabel")
    nameLabel.Size = UDim2.new(0, 100, 0, 20)
    nameLabel.Position = UDim2.new(0.5, -50, 0.1, 0)
    nameLabel.Text = player.Name
    nameLabel.Parent = ScreenGui

    -- Tạo thanh máu
    local healthBar = Instance.new("Frame")
    healthBar.Size = UDim2.new(0, 100, 0, 10)
    healthBar.Position = UDim2.new(0.5, -50, 0.2, 0)
    healthBar.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    healthBar.Parent = ScreenGui

    -- Tạo dây màu trắng định vị người chơi
    local line = Instance.new("Line")
    line.Thickness = 2
    line.Color = Color3.fromRGB(255, 255, 255)
    line.Parent = ScreenGui

    -- Hàm để cập nhật ESP
    local function updateESP()
        -- Cập nhật tên người chơi
        nameLabel.Text = player.Name

        -- Cập nhật thanh máu
        healthBar.Size = UDim2.new(0, player.Character.Humanoid.Health / player.Character.Humanoid.MaxHealth * 100, 0, 10)

        -- Cập nhật dây màu trắng định vị người chơi
        line.From = Vector2.new(game.Players.LocalPlayer.Character.HumanoidRootPart.Position.X, game.Players.LocalPlayer.Character.HumanoidRootPart.Position.Y)
        line.To = Vector2.new(player.Character.HumanoidRootPart.Position.X, player.Character.HumanoidRootPart.Position.Y)
        -- Tạo dây màu trắng định vị người chơi
